        // Web Audio Calming Frequency Generator (Solfeggio 432Hz Ambient Pad)
        let audioCtx = null;
        let osc1 = null;
        let osc2 = null;
        let gainNode = null;
        let filter = null;
        let isAudioPlaying = false;

        function toggleHealingSound() {
            const btnIcon = document.getElementById('audio-icon');
            const statusText = document.getElementById('audio-status');
            const pulse = document.getElementById('audio-pulse');

            if (!isAudioPlaying) {
                if (!audioCtx) {
                    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                }
                
                if (audioCtx.state === 'suspended') {
                    audioCtx.resume();
                }

                osc1 = audioCtx.createOscillator();
                osc2 = audioCtx.createOscillator();
                gainNode = audioCtx.createGain();
                filter = audioCtx.createBiquadFilter();

                osc1.type = 'sine';
                osc1.frequency.setValueAtTime(108, audioCtx.currentTime); // Deep octave of 432
                
                osc2.type = 'triangle';
                osc2.frequency.setValueAtTime(108.5, audioCtx.currentTime); // chorus beat effect

                filter.type = 'lowpass';
                filter.frequency.setValueAtTime(250, audioCtx.currentTime);

                gainNode.gain.setValueAtTime(0, audioCtx.currentTime);
                gainNode.gain.linearRampToValueAtTime(0.18, audioCtx.currentTime + 3);

                osc1.connect(filter);
                osc2.connect(filter);
                filter.connect(gainNode);
                gainNode.connect(audioCtx.destination);

                osc1.start();
                osc2.start();

                isAudioPlaying = true;
                btnIcon.setAttribute('data-lucide', 'square');
                statusText.innerText = 'Activo';
                pulse.style.display = 'inline-block';
            } else {
                if (gainNode) {
                    gainNode.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 1.2);
                    setTimeout(() => {
                        try {
                            osc1.stop();
                            osc2.stop();
                        } catch(e) {}
                    }, 1300);
                }
                isAudioPlaying = false;
                btnIcon.setAttribute('data-lucide', 'play');
                statusText.innerText = '432Hz';
                pulse.style.display = 'none';
            }
            lucide.createIcons();
        }

        let breathInterval = null;
        function startBreathingCircleGuide() {
            if(breathInterval) clearInterval(breathInterval);
            const phases = [
                { text: 'Inhala...', duration: 4 },
                { text: 'Retén...', duration: 4 },
                { text: 'Exhala...', duration: 4 },
                { text: 'Espera...', duration: 4 }
            ];
            let currentIdx = 0;
            let stepSec = 0;

            const labelEl = document.getElementById('breath-text');
            labelEl.innerText = phases[0].text;

            breathInterval = setInterval(() => {
                stepSec++;
                if(stepSec >= phases[currentIdx].duration) {
                    stepSec = 0;
                    currentIdx = (currentIdx + 1) % phases.length;
                    labelEl.innerText = phases[currentIdx].text;
                }
            }, 1000);
        }

        // 15 Min Protocol Controller logic
        function toggleProtocolTimer() {
            const playBtn = document.getElementById('btn-play-protocolo');
            if (!isProtocolRunning) {
                isProtocolRunning = true;
                playBtn.innerHTML = '<i data-lucide="pause" class="w-4 h-4 text-slate-950 fill-slate-950"></i>';
                protocolTimer = setInterval(() => {
                    if (protocolSeconds > 0) {
                        protocolSeconds--;
                        updateTimerUI();
                        autoPhaseSwitchCheck();
                    } else {
                        clearInterval(protocolTimer);
                        isProtocolRunning = false;
                        playBtn.innerHTML = '<i data-lucide="play" class="w-4 h-4 text-slate-950 fill-slate-950"></i>';
                        showSuccessToast("¡Protocolo completado! ¡Tu mente está en paz!");
                    }
                }, 1000);
            } else {
                isProtocolRunning = false;
                clearInterval(protocolTimer);
                playBtn.innerHTML = '<i data-lucide="play" class="w-4 h-4 text-slate-950 fill-slate-950"></i>';
            }
            lucide.createIcons();
        }

        function updateTimerUI() {
            const minutes = Math.floor(protocolSeconds / 60);
            const seconds = protocolSeconds % 60;
            document.getElementById('protocolo-time').innerText = 
                `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
        }

        function autoPhaseSwitchCheck() {
            let idealPhase = 1;
            if (protocolSeconds < 5 * 60) {
                idealPhase = 3;
            } else if (protocolSeconds < 12 * 60) {
                idealPhase = 2;
            }

            if (idealPhase !== currentPhase) {
                currentPhase = idealPhase;
                updateProtocolPhaseUI();
            }
        }

                    <div class="text-[9px] text-red-400/70 line-through font-semibold max-w-[90%] truncate">${item.lie}</div>
                    <div class="text-xs font-bold text-emerald-400 italic leading-tight">${item.truth}</div>
                    <div class="text-[8px] text-slate-500 font-bold uppercase tracking-widest mt-0.5">${item.ref}</div>
                </div>
            `).join('');
            lucide.createIcons();
        }

        // TAB 5: Daily Gratitude Logic
        function saveGratitude() {
            const g1 = document.getElementById('gratitude-1').value.trim();
            const g2 = document.getElementById('gratitude-2').value.trim();
            const g3 = document.getElementById('gratitude-3').value.trim();

            if (!g1 || !g2 || !g3) {
                showToast("Completa los tres campos para registrar.", "warning");
                return;
            }

            let list = JSON.parse(localStorage.getItem('gratitude_list') || '[]');
            list.unshift({
                items: [g1, g2, g3],
                date: new Date().toLocaleDateString('es-ES'),
                id: Date.now()
            });

            localStorage.setItem('gratitude_list', JSON.stringify(list));

            document.getElementById('gratitude-1').value = '';
            document.getElementById('gratitude-2').value = '';
            document.getElementById('gratitude-3').value = '';

            renderGratitudes();
            showSuccessToast("¡Agradecimientos registrados!");
        }

        function renderGratitudes() {
            const list = JSON.parse(localStorage.getItem('gratitude_list') || '[]');
            const container = document.getElementById('gratitude-wall');
            if (list.length === 0) {
                container.innerHTML = `
                    <div class="py-4 text-center text-[10px] text-slate-500 border border-dashed border-slate-800 rounded-xl">
                        Aún no tienes registros de agradecimiento.
                    </div>
                `;
                return;
            }

            container.innerHTML = list.map(item => `
                <div class="bg-slate-900/40 border border-slate-900 p-3.5 rounded-xl space-y-1 relative">
                    <button onclick="deleteGratitude(${item.id})" class="absolute top-2 right-2 text-slate-600 hover:text-red-400 transition">
                        <i data-lucide="x" class="w-3.5 h-3.5"></i>
                    </button>
                    <div class="text-[8px] font-black text-emerald-400 uppercase tracking-widest mb-1">${item.date}</div>
                    <ul class="space-y-0.5 list-disc list-inside">
                        ${item.items.map(txt => `<li class="text-xs text-slate-300 font-semibold truncate">${txt}</li>`).join('')}
                    </ul>
                </div>
            `).join('');
            lucide.createIcons();
        }

        function deleteTruth(id) {
            let list = JSON.parse(localStorage.getItem('truths_list') || '[]');
            list = list.filter(item => item.id !== id);
            localStorage.setItem('truths_list', JSON.stringify(list));
            renderSavedTruths();
        }

        function deleteGratitude(id) {
            let list = JSON.parse(localStorage.getItem('gratitude_list') || '[]');
            list = list.filter(item => item.id !== id);
            localStorage.setItem('gratitude_list', JSON.stringify(list));
            renderGratitudes();
        }

            if(confirm("¿Seguro que deseas borrar todos tus diarios y actividades guardadas en el dispositivo?")) {
                localStorage.clear();
                switchTab(activeTab);
                showSuccessToast("Datos locales restablecidos.");
            }
        }
